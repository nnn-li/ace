/**
 * @param {string} s
 */
export function floatAST(s: string) {
    var thing: {
        text: string;
        value: number;
        isFloat: () => boolean;
        isInt: () => boolean;
        isLong: () => boolean;
        toString: () => string;
    } = {
            text: s,
            value: parseFloat(s),
            isFloat: function() { return true; },
            isInt: function() { return false; },
            isLong: function() { return false; },
            toString: function() {return s }
        };
    return thing;
}
